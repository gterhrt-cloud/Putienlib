-- Slider System
local function CreateSlider(parent, title, min, max, default, callback)
    local SliderContainer = Instance.new("Frame")
    SliderContainer.Name = title .. "SliderContainer"
    SliderContainer.Size = UDim2.new(1, 0, 0, 70)
    SliderContainer.BackgroundTransparency = 1
    SliderContainer.Parent = parent

    local TitleLabel = Instance.new("TextLabel")
    TitleLabel.Name = "Title"
    TitleLabel.Size = UDim2.new(1, -20, 0, 20)
    TitleLabel.Position = UDim2.new(0, 10, 0, 0)
    TitleLabel.BackgroundTransparency = 1
    TitleLabel.Text = title
    TitleLabel.TextColor3 = Color3.fromRGB(180, 180, 220)
    TitleLabel.TextSize = 12
    TitleLabel.Font = Enum.Font.GothamSemibold
    TitleLabel.TextXAlignment = Enum.TextXAlignment.Left
    TitleLabel.Parent = SliderContainer

    local SliderBackground = Instance.new("Frame")
    SliderBackground.Name = "SliderBg"
    SliderBackground.Size = UDim2.new(1, -20, 0, 6)
    SliderBackground.Position = UDim2.new(0, 10, 0, 28)
    SliderBackground.BackgroundColor3 = Color3.fromRGB(35, 45, 75)
    SliderBackground.BorderSizePixel = 0
    SliderBackground.Parent = SliderContainer

    local SliderCorner = Instance.new("UICorner")
    SliderCorner.CornerRadius = UDim.new(0, 3)
    SliderCorner.Parent = SliderBackground

    local SliderFill = Instance.new("Frame")
    SliderFill.Name = "SliderFill"
    SliderFill.Size = UDim2.new((default - min) / (max - min), 0, 1, 0)
    SliderFill.BackgroundColor3 = Color3.fromRGB(100, 200, 255)
    SliderFill.BorderSizePixel = 0
    SliderFill.Parent = SliderBackground

    local FillCorner = Instance.new("UICorner")
    FillCorner.CornerRadius = UDim.new(0, 3)
    FillCorner.Parent = SliderFill

    local ValueLabel = Instance.new("TextLabel")
    ValueLabel.Name = "Value"
    ValueLabel.Size = UDim2.new(0, 50, 0, 20)
    ValueLabel.Position = UDim2.new(1, -60, 0, 24)
    ValueLabel.BackgroundTransparency = 1
    ValueLabel.Text = tostring(math.floor(default))
    ValueLabel.TextColor3 = Color3.fromRGB(100, 200, 255)
    ValueLabel.TextSize = 12
    ValueLabel.Font = Enum.Font.GothamBold
    ValueLabel.Parent = SliderContainer

    local dragging = false
    local currentValue = default

    SliderBackground.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            local mousePos = input.Position
            local sliderPos = SliderBackground.AbsolutePosition.X
            local sliderSize = SliderBackground.AbsoluteSize.X
            local relativePos = math.clamp(mousePos.X - sliderPos, 0, sliderSize)
            local percentage = relativePos / sliderSize

            currentValue = math.floor(min + (max - min) * percentage)
            currentValue = math.clamp(currentValue, min, max)

            SliderFill.Size = UDim2.new(percentage, 0, 1, 0)
            ValueLabel.Text = tostring(currentValue)

            if callback then
                callback(currentValue)
            end
        end
    end)

    return {
        SetValue = function(value)
            currentValue = value
            local percentage = (value - min) / (max - min)
            SliderFill.Size = UDim2.new(percentage, 0, 1, 0)
            ValueLabel.Text = tostring(math.floor(value))
        end
    }
end

-- Button Function
local function CreateButton(parent, text, callback)
    local Button = Instance.new("TextButton")
    Button.Name = text .. "Button"
    Button.Size = UDim2.new(1, -20, 0, 40)
    Button.Position = UDim2.new(0, 10, 0, 0)
    Button.BackgroundColor3 = Color3.fromRGB(35, 45, 75)
    Button.BorderSizePixel = 0
    Button.Text = text
    Button.TextColor3 = Color3.fromRGB(100, 200, 255)
    Button.TextSize = 13
    Button.Font = Enum.Font.GothamSemibold
    Button.Parent = parent

    local ButtonCorner = Instance.new("UICorner")
    ButtonCorner.CornerRadius = UDim.new(0, 8)
    ButtonCorner.Parent = Button

    local hoverIn = TweenService:Create(Button, TweenInfo.new(0.2), { BackgroundColor3 = Color3.fromRGB(60, 100, 150) })
    local hoverOut = TweenService:Create(Button, TweenInfo.new(0.2), { BackgroundColor3 = Color3.fromRGB(35, 45, 75) })

    Button.MouseEnter:Connect(function() hoverIn:Play() end)
    Button.MouseLeave:Connect(function() hoverOut:Play() end)
    Button.MouseButton1Click:Connect(callback)

    return Button
end

-- Drag functionality
local dragging = false
local dragStartPos = nil
local lastMousePos = nil

TitleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStartPos = input.Position
        lastMousePos = input.Position
    end
end)

TitleBar.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        lastMousePos = input.Position
    end
end)

local currentTween = nil
game:GetService("RunService").RenderStepped:Connect(function()
    if dragging and lastMousePos then
        local delta = lastMousePos - dragStartPos
        local targetPos = MainContainer.Position + UDim2.new(0, delta.X, 0, delta.Y)

        if currentTween then
            currentTween:Cancel()
        end

        currentTween = TweenService:Create(
            MainContainer,
            TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
            { Position = targetPos }
        )
        currentTween:Play()
        dragStartPos = lastMousePos
    end
end)

-- Boot sequence
local function Boot()
    local bootSteps = {
        { text = "Initializing Core...", duration = 0.6 },
        { text = "Loading UI Framework...", duration = 0.7 },
        { text = "Applying Theme...", duration = 0.6 },
        { text = "Optimizing Performance...", duration = 0.5 },
        { text = "Ready", duration = 0.4 }
    }

    local totalDuration = 0
    for _, step in ipairs(bootSteps) do
        totalDuration = totalDuration + step.duration
    end

    local elapsedTime = 0

    for _, step in ipairs(bootSteps) do
        BootStatus.Text = step.text

        local progressAnim = TweenService:Create(
            ProgressBar,
            TweenInfo.new(step.duration, Enum.EasingStyle.Linear),
            { Size = UDim2.new(elapsedTime / totalDuration + step.duration / totalDuration, 0, 1, 0) }
        )
        progressAnim:Play()

        elapsedTime = elapsedTime + step.duration
        task.wait(step.duration)
    end

    local fadeAnim = TweenService:Create(
        BootScreen,
        TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.In),
        { BackgroundTransparency = 1 }
    )
    fadeAnim:Play()

    task.wait(0.3)
    BootScreen:Destroy()
end
